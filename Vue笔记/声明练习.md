# 声明练习

```js
const a = function (a, b) {
  return a + b;
};
// const c = a(1, 2);
let c;
c = a(1, 2);
```

```js
const c = {
  a: "abc",
  b: 2,
};
const a = function (c) {
  return c.a + c.b;
};
a(c);
```

```js
const e = [0, 1, 2];
const d = [0, 1, 2];
const a = function (e, d) {
  return e[0] + e[1] + e[2] + d[0] + d[1] + d[2];
};
a(e, d);
```

```js
const e = [0, 1, 2];
const a = function (e) {
  let sum = 0;
  for (const item of e) {
    sum = sum + item;
  }
  return sum;
};
a(e);
```
