# 声明练习

```js
const a = function (a, b) {
  return a + b;
};
// const c = a(1, 2);
let c;
c = a(1, 2);
```

```js
const c = {
  a: "abc",
  b: 2,
};
const a = function (c) {
  return c.a + c.b;
};
a(c);
```

```js
const e = [0, 1, 2];
const d = [0, 1, 2];
const a = function (e, d) {
  return e[0] + e[1] + e[2] + d[0] + d[1] + d[2];
};
a(e, d);
```

```js
const e = [0, 1, 2];
const a = function (e) {
  let sum = 0;
  for (const item of e) {
    sum = sum + item;
  }
  return sum;
};
a(e);
```

## 属性的遍历：for...in 循环,for...of 循环

```js
// for...in循环用来遍历一个对象的全部属性。
var obj = { a: 1, b: 2, c: 3 };
// i 表示的是对象属性，obj是遍历的对象
for (var i in obj) {
  console.log("键名：", i);
  console.log("键值：", obj[i]);
}
// 键名： a
// 键值： 1
// 键名： b
// 键值： 2
// 键名： c
// 键值： 3

// for...of循环借助Object.keys方法用来遍历一个对象的全部属性。
const obj = {
  a: "111",
  b: "222",
  c: 333,
};

for (const i of Object.keys(obj)) {
  console.log(obj[i]);
}
```
